	\posttitle{\end{flushleft}}
	\preauthor{\begin{flushleft} \large \vspace{-0.1cm} }
	\author{\aname}
	\postauthor{\end{flushleft}}
	\predate{\begin{flushleft} \vspace{-0.1cm}}
	\date{\today}
	\postdate{\end{flushleft} \vspace{-0.5cm}}
	
	% FIRST PAGE
	\begin{minipage}[c]{0.08\textwidth}

	\maketitle
	\end{minipage}
	\thispagestyle{empty}
}
